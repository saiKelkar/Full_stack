function maskEmail(email) {
  let hash = email.indexOf("@");
  let mainEmail = email.slice(1, hash-1)
  let maskEmail = email.replace(mainEmail, "*".repeat(mainEmail.length))
  return maskEmail;
}

let email = "kelkar.sai99@gmail.com";
console.log(maskEmail(email));
